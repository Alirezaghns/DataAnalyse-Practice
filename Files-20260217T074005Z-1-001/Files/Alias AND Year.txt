/****** Script for SelectTopNRows command from SSMS  ******/

USE Northwind
SELECT FirstName + ' ' + LastName AS FULLNAME
, BirthDate , YEAR(BirthDate) AS YearBirthDate , YEAR(GETDATE()) AS TODAYEDATE
, (YEAR(GETDATE()) - YEAR(BirthDate)) AS AGE
FROM Employees
WHERE (YEAR(GETDATE()) - YEAR(BirthDate)) < 70
ORDER BY  AGE , FULLNAME
