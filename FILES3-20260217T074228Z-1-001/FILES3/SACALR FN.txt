CREATE FUNCTION udf_GerOrderTotal (@OrderID INT)
RETURNS MONEY
AS
BEGIN
DECLARE @TOTAL Money;
SELECT @TOTAL = SUM (UnitPrice  Quantity)
FROM [Order Details]
WHERE OrderID = @OrderID
RETURN @TOTAL
END
GO
