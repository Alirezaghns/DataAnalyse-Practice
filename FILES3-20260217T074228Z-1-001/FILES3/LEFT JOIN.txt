SELECT c.CompanyName , MAX(o.OrderDate) AS LASTORDER
FROM Customers c
LEFT JOIN Orders o ON o.CustomerID = c.CustomerID
--WHERE o.CustomerID IS NULL
GROUP BY c.CompanyName
