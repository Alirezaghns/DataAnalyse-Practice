CREATE PROCEDURE  usp_GetCustomerByCountry
@Country Nvarchar(50)
AS
BEGIN
SELECT CustomerID , CompanyName , City , Country
FROM Customers
WHERE Country = @Country
ORDER BY CompanyName
END
GO

EXEC usp_GetCustomerByCountry @Country = N'Germany'
